📘Modelos Probabilísticos

\textbf{🧠 Idea Principal}
\text{La incertidumbre es inherente a muchos fenómenos reales: mediciones, decisiones, procesos}
\text{naturales y sociales. Los modelos deterministas no bastan.}
\text{Necesitamos herramientas que permitan describir, cuantificar y predecir comportamientos}
\text{aleatorios.}
\text{📌 Ejemplos de situaciones aleatorias}
\text{🔹 ¿Cuánto tiempo pasará hasta que llegue el próximo bus?}
\text{🔹 ¿Cuántos clientes llegarán a una tienda en una hora?}
\text{🔹 ¿Cuál será el retorno de una acción financiera mañana?}
\text{🔹 ¿Cuál es la probabilidad de que un paciente tenga la enfermedad dado un test?}

📘Motivación

\textbf{🧩 Motivación formal}
\text{Un modelo probabilístico es un puente entre la teoría de la probabilidad y los datos reales.}
\text{Nos permite representar sistemas con variables aleatorias, distribuciones, y relaciones}
\text{matemáticas coherentes.}
\textbf{🔧 Necesidad de un modelo}
\text{✅ Hacer predicciones}
\text{No basta con decir que un resultado es “aleatorio”. Se necesita una representación}
\text{matemática que nos permita:}
\text{✅ Calcular probabilidades}
\text{✅ Estimar parámetros}
\text{✅ Tomar decisiones óptimas}

📘Ensayo de Bernoulli

\textbf{📌 ¿Qué es un ensayo de Bernoulli?}
\text{Es un experimento aleatorio en el que sólo se pueden obtener dos resultados etiquetados}
\left\{\begin{matrix} \text{éxito} &\rightarrow 1 \\ \text{ fracaso} &\rightarrow 0 \end{matrix}\right.
\text{usualmente como éxito y fracaso.}
\text{Ensayo de Bernoulli} \Rightarrow
\textbf{🧩 Este modelo representa situaciones como:}
\text{🔹 Lanzamiento de una moneda (cara/cruz),}
\text{🔹 Resultado positivo/negativo de una prueba médica,}
\text{🔹 Detección/pérdida de un paquete en una red.}
\text{Utilizaremos este experimento/ensayo para definir diferentes distribuciones.}
\text{Denotaremos por} \ p \text{ a la probabilidad de obtener éxito y entonces } p \in (0,1).

📘Distribución de Bernoulli

\textbf{📌 Definición}
\text{Sea} \ X \ \text{una variable aleatoria discreta tal que:}
X = \left\{\begin{matrix} 1, & \text{si ocurre un éxito} \\ 0, & \text{si ocurre un fracaso} \end{matrix}\right.
🔹 \ f(x) = \mathbb{P}(X = x)=p^x(1-p)^{1-x}; \quad x = 0,1.
\text{Entonces se dice que} \ X \sim\text{Bernoulli}(p) \ \text{donde }p = \mathbb{P}(X=1), \ \text{es decir, } p \ \text{es la probabilidad}
\text{de tener éxito en el experimento.}
🔹 \ \mathbb{E}[X]=p.
🔹 \operatorname{Var}(X)=p(1-p).
\textbf{🧠 Función de probabilidad, Valor esperado y Varianza}
\textbf{🧩 Ejemplo: Diagnóstico médico}
\text{Una prueba rápida tiene } p = 0.85 \ \text{de detectar correctamente la enfermedad (sensibilidad).}
\text{Si definimos }X=1 \ \text{si la prueba detecta la enfermedad y }X=0 \ \text{si no, entonces:}
X \sim \mathrm{Bernoulli}(0.85), \quad \mathbb{E}[X] = 0.85, \quad \operatorname{Var}(X) = 0.1275.

📘Distribución Binomial

\textbf{📌 Definición}
\text{Sea} \ X \ \text{una variable aleatoria que cuenta el } \textbf{número de éxitos en }n \ \text{ensayos independientes}
X \sim \text{Bin}(n,p).
🔹 \ f(x) = \mathbb{P}(X = x)=\binom{n}{x}p^x(1-p)^{n-x}; \quad x = 0,1,\ldots,n.
\text{de Bernoulli con probabilidad de éxito} \ p. \ \text{Entonces se dice que,}
\text{Es decir,} \ X \ \text{sigue una distribución binomial con parámetros} \ n,p.
🔹 \ \mathbb{E}[X]=np.
🔹 \operatorname{Var}(X)=np(1-p).
\textbf{🧠 Función de probabilidad, Valor esperado y Varianza}
\textbf{🧩 Ejemplo:}
\text{En un proceso de producción, cada unidad tiene probabilidad }p=0.05 \ \text{de ser defectuosa.}
\text{En un lote de,} \ n=20 \ \text{unidades, sea }X \ \text{el número de defectuosos:} \ X \sim \mathrm{Bin}(20, 0.05),
\mathbb{E}[X] = 1, \quad \operatorname{Var}(X) = 0.95, \quad \mathbb{P}(X=0)=(0.95)^20\approx 0.358.

📘Distribución Geométríca

\textbf{📌 Definición}
\text{Sea} \ X \ \text{una variable aleatoria que cuenta el } \textbf{número de fracasos antes del primer éxito}
X \sim \text{Geom}(p).
🔹 \ f(x) = \mathbb{P}(X = x)=p(1-p)^x; \quad x = 0,1,2,\ldots
\text{en una secuencia infinita de ensayos de Bernoulli independientes, con tasa de éxito} \ p.
\text{Entonces, se dice que }
🔹 \ \mathbb{E}[X]=\frac{1-p}{p}.
🔹 \operatorname{Var}(X)=\frac{1-p}{p^2}.
\textbf{🧠 Función de probabilidad, Valor esperado y Varianza}
\textbf{🧩 Ejemplo:}
\text{Una red digital transmite paquetes con probabilidad de éxito }p=0.2. \ \text{Sea } X \text{el número de}
\text{fallos antes de un paquete exitoso:}
X \sim \text{Geom}(0.2), \quad \mathbb{E}[X] = 4, \quad \operatorname{Var}(X) = 20.

📘Distribución Binomial             Negativa

\textbf{📌 Definición}
\text{Sea} \ X \ \text{una variable aleatoria que representa el } \textbf{número de fracasos antes de observar}
X \sim \text{NegBin}(r, p).
🔹 \ f(x) = \mathbb{P}(X = x)=\binom{x+r-1}{x}p^r(1-p)^x; \quad x = 0,1,2,\ldots
r \ \textbf{éxitos } \text{en una secuencia infinita de ensayos de Bernoulli independientes, con tasa de éxito}
p. \ \text{Entonces, se dice que }
🔹 \ \mathbb{E}[X]=r\frac{(1-p)}{p}.
🔹 \operatorname{Var}(X)=r\frac{(1-p)}{p^2}.
\textbf{🧠 Función de probabilidad, Valor esperado y Varianza}
\textbf{🧩 Ejemplo:}
\text{En un experimento biológico, se repite un proceso hasta observar } r=3 \ \text{éxitos. Cada intento}
\text{tiene éxito con probabilidad} \ p=0.25. \ \text{Sea} \ X \ \text{el número de fracasos antes del tercer éxito:}
X \sim \text{NegBin}(3,0.25), \quad \mathbb{E}[X] = 9, \quad \operatorname{Var}(X) = 36.

📘Distribución Hipergeométrica

\textbf{📌 Definición}
\text{Sea} \ X \ \text{una variable aleatoria que cuenta el } \textbf{el número de éxitos obtenidos al extraer}
X \sim \text{HiperGeom}(N,K,n).
🔹 \ f(x) = \mathbb{P}(X = x)=\frac{\binom{K}{x}\binom{N-K}{n-x}}{\binom{N}{n}}; \quad x \in \left\{ \max(0, n - N + K), \ldots, \min(n, K) \right\}.
\textbf{sin reemplazo} \ n \ \text{elementos de una población finita de tamaño} \ N, \ \text{que contiene exacta-}
\text{mente }K \ \text{elementos exitosos. Entonces se dice que}
🔹 \ \mathbb{E}[X]=n \cdot \frac{K}{N}
🔹 \operatorname{Var}(X)=n \cdot \frac{K}{N} \cdot \left(1 - \frac{K}{N} \right) \cdot \frac{N - n}{N - 1}
\textbf{🧠 Función de probabilidad, Valor esperado y Varianza}
\textbf{🧩 Ejemplo:}
\text{Una caja contiene }N=20 \ \text{chips, de los cuales }K=5 \ \text{son defectuosos. Se seleccionan } n=4
\text{al azar sin reemplazo. Sea }X \ \text{el número de defectuosos en la muestra:}
X \sim \text{HiperGeom}(20,5,4), \quad \mathbb{E}[X] =4 \cdot \frac{5}{20} = 1, \quad \operatorname{Var}(X) = \frac{12}{19}.

📘Distribución Exponencial

\textbf{📌 Definición}
\text{Sea} \ X \ \text{una variable aleatoria continua que representa el }\textbf{tiempo hasta la ocurrencia}
🔹 \ f(x) = \lambda e^{-\lambda x}; \quad x\geq 0.

X \sim \mathrm{Exponencial}(0.5), \quad \mathbb{E}[X] = 2, \quad \operatorname{Var}(X) = 4.

📘Distribución Beta

🔹 \ \mathbb{E}[X]=\mu
🔹 \operatorname{Var}(X)=\sigma^2

📘Normal Estándar

\textbf{📌 Definición}
\text{La normal estándar es} \ Z \sim \mathcal{N}(0,1), \ \text{con densidad:}
\text{Si} \ X \sim \mathcal{N}(\mu, \sigma^2), \ \text{entonces:}
\phi(z) = \frac{1}{\sqrt{2\pi}} e^{-z^2/2}.
Z = \frac{X - \mu}{\sigma} \sim \mathcal{N}(0, 1).
\text{Esta transformación se denomina tipificación o estandarización, y permite trabajar con}
\text{tablas universales de la normal.}
\textbf{🧩 Ejemplo:}
\text{Los pesos de cierto tipo de batería siguen }X \sim \mathcal{N}(120, 9). \ \text{La probabilidad de que pese menos}
\text{de 117 gramos es:}
Z = \frac{117 - 120}{3} = -1 \Rightarrow \mathbb{P}(X < 117) = \mathbb{P}(Z < -1) \approx 0.1587.
\text{🔹 Límite de distribuciones de sumas: Teorema Central del Límite}
\text{🔹 Aproximación a binomial y Poisson (bajo ciertas condiciones)}

📘Normal Estándar

\text{Muchos modelos discretos o complejos pueden ser aproximados por otros más simples o }
\text{continuos bajo ciertas condiciones. Estas aproximaciones permiten facilitar cálculos analíticos}
\text{y estadísticos, especialmente en contextos asintóticos o de gran tamaño.}
\text{🔸 Aproximaciones clásicas.}
Aproximación Condiciones principales Resultado

Poisson → Normal

Teorema Central del Límite Sumas de variables i.i.d.
n \to \infty, \ p\to 0, \ np = \lambda
n \to \infty, \ \text{con } 0 < p < 1

\text{Bin}(n,p)\approx \text{Poisson}(\lambda)
\mathrm{Bin}(n,p) \approx \mathcal{N}(np, np(1-p))
\mathrm{Poisson}(\lambda) \approx \mathcal{N}(\lambda, \lambda)
\mathrm{Gamma}(\alpha, \beta) \approx \mathcal{N}(\mu, \sigma^2)

\text{cuando el modelo exacto es incómodo o costoso de calcular directamente.}
