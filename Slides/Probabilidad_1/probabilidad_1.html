📘Operaciones con Eventos

\text{🟦 Supuestos}
\text{Sea} \ \Omega \ \text{un espacio muestral y } A,B \in \Omega \ \text{dos eventos.}
\text{🟩 Operaciones básicas}
\text{1. Unión de eventos:}
A \cup B = \{\omega \in \Omega : \omega \in A \text{ o } \omega \in B\}
\text{2. Intersección de eventos:}
A \cap B = \{\omega \in \Omega : \omega \in A \text{ y } \omega \in B\}
\text{Evento que ocurre si } \textbf{ocurre A o B o ambos.}
\text{Evento que ocurre si } \textbf{ocurren A y B simultáneamente}
\text{3. Complemento de un evento:}
A^c = \{\omega \in \Omega : \omega \notin A\}
\text{Evento que ocurre si } \textbf{no ocurre A.}

📘Operaciones con Eventos

\text{📊 Ejemplo}
\text{Supón que se lanza un dado,} \ \Omega = \{1,2,3,4,5,6\}
\text{Sean}
\text{🔹 } A = \{\text{número par}\} = \{2,4,6\}
\text{🔹 } B=\{\text{número} > 3\} = \{4,5,6\}
Operación Resultado
A \cup B
\{2,4,5,6\}
A \cap B
\{4,6\}
A^c
\{1,3,5\}
\text{🧩 Propiedades lógicas (mención breve)}
\text{🔹 Conmutatividad: } A \cup B = B \cup A, A\cap B = B \cap A
\text{🔹 Distributividad: } A \cup (B \cap C) = (A \cup B) \cap (A\cup C)
\text{🔹 Distributividad: } (A \cup B)^c = A^c \cap B^c, (A \cap B)^c = A^c \cup B^c

📘Enfoques de la Probabilidad

\textbf{🟩 1. Probabilidad Clásica o de Laplace}
\text{Se basa en el principio de equiprobabilidad. Si todos los resultados posibles son}
P(A)=\frac{|A|}{|\Omega|}=\frac{\text{número de casos favorables}}{\text{número total de casos posibles}}
\text{igualmente probables:}
\text{🔹 Aplica a espacios finitos y simétricos}
\textbf{🧪 Ejemplo}
\text{Lanzar dos dados distinguibles}
\Omega = \{(x,y): x,y = 1,2, \ldots, 6\}
A = \text{Ambos dados muestran el mismo número}
B = \text{Al menos un dado cae 6}
P(A)=\frac{6}{36}=\frac{1}{6}
P(B)=\frac{11}{36}
\text{✅ Intuitiva y sencilla}
\text{❌ No sirve si los resultados no son equiprobables o si el espacio es continuo}

📘Enfoques de la Probabilidad

\textbf{🟦 2. Probabilidad Geométrica}
\text{Se usa cuando el espacio de resultados es continuo y se puede medir (longitud,}
P(A)=\frac{\text{medida (longitud, área, etc) de A}}{\text{medida de }\Omega}
\text{área, volumen).}
\textbf{🧪 Ejemplo}
\text{¿cuál es la probabilidad de que un punto elegido al azar en [0,1] caiga en [0.3, 0.6]?}
P=\frac{0.6-0.3}{1-0}=0.3
\text{✅ Introduce el concepto de “medida”}
\text{❌ Ambigua si no se define claramente la noción de “azar” (paradojas como la de}
\text{Bertrand)}

📘Enfoques de la Probabilidad

\textbf{🟥 3. Probabilidad Frecuentista}
\text{La probabilidad de un evento es el límite de su frecuencia relativa en un número}
\text{grande de repeticiones del experimento:}
\text{✅ Aplicable en el laboratorio o experimentación}
\text{❌ No aplica bien a eventos únicos o no repetibles}
P(A)=\lim_{n \to \infty}\frac{nA}{n}
\text{🔹 Intérprete operativo y empírico}
\text{🔹 Base de la estadística clásica}
\text{❌ No es una definición matemática formal, depende de un límite}

📘Enfoques de la Probabilidad

\textbf{🟨 4. Probabilidad Axiomática}
\text{Define la probabilidad como una medida} \ P \ \text{sobre una } \sigma-\text{álgebra de eventos,}
\text{cumpliendo tres axiomas.}
\text{✅ General, rigurosa, coherente}
\text{✅ Conecta con teoría de la medida}
\text{🔹 No depende del tipo de espacio (finito, infinito, continuo)}
\text{🔹 Es la base moderna de la teoría matemática de la probabilidad}
\text{✅ Permite demostrar teoremas}
Enfoque Tipo de espacio Intuitivo Rigor
 		 Limitaciones
Clásico Finito y simétrico ✅ ❌ No aplicable a casos asimétricos o infinitos
Geométrico Continuo medible ✅ ⚠️ Ambiguo sin definición precisa de medida

📘Espacio de Probabilidad y                        Axiomas de Kolmogórov

\textbf{🧩 Definición: Espacio de probabilidad}
\text{Un espacio de probabilidad es una terna:} \ (\Omega, \mathcal{A}, P) \ \text{donde:}
\text{🔹 }\Omega \ \text{conjunto no vacío llamado espacio muestra}
\text{🔹} \ \mathcal{A} \subseteq P(A):\sigma-\text{álgebra de conjuntos de }\Omega
\text{🔹} \ P:\mathcal{A} \to [0,1]: \ \text{función llamada probabilidad}
\textbf{🧱 Axiomas de Kolmogórov}
\text{Sea} \ A \subseteq \Omega \ \text{un evento. La función} \ P \ \text{debe satisfacer:}
\text{🔹 }0 \leq P(A) \leq 1, \quad \forall A \in \mathcal{A}
\text{🔹} \ P(\Omega)=1
\text{🔹} \ \text{Si} \ A_1, A_2, \dots \in \mathcal{A} \ \text{y} \ A_i \cap A_j = \emptyset \ \text{para} \ i \ne j, \text{entonces:}
P\left( \bigcup_{i=1}^\infty A_i \right) = \sum_{i=1}^\infty P(A_i)

📘Algunas propiedades

\textbf{📌 Proposición}
\text{Sean} \ A,B \subseteq \Omega, \ \text{con} \ A,B \in \mathcal{A}, \ \text{entonces, se cumplen las siguientes propiedades:}
i) \ P(A) = 1- P(A^c), \quad P(A^c) = 1-P(A)
ii) \ P(\emptyset) = 0
iii) \ \text{Si} \ A \subseteq B, \ \text{entonces} \ P(A) \leq P(B)
\textbf{🧩 Observación}
\text{Si} \ A \cap B = \emptyset, \ \text{entonces}
iv) \ P(A\cup B) = P(A) + P(B) - P(A\cap B)
P(A\cup B) = P(A)+P(B)
\text{Es un caso particular de la regla general, y coincide con el axioma de aditividad.}

📘Probabilidad Condicional

\textbf{📌 Definición}
\text{Sean} \ A,B \in \mathcal{A}, \ \text{con} \ P(B)>0. \ \text{La probabilidad condicional de} \ A \ \text{dado} \ B \ \text{se}
\text{define como:}
P(A|B) = \frac{P(A\cap B)}{P(B)}
\textbf{🔎 Interpretación}
\text{🔹 }P(A∣B) \ \text{representa la probabilidad de que ocurra} \ A, \ \text{asumiendo que ya}
\text{ocurrió} \ B.
\text{🔹 }\text{Es una nueva función de probabilidad, restringida al subconjunto} \ B.
\textbf{📊 Ejemplo:}
\text{Se lanza un dado al aire, y definimos}
\text{🔹 }A=\{\text{número par}\}=\{2,4,6\}
\text{🔹 }B=\{\text{número mayor que }3\}=\{4,5,6\}
\text{entonces,} \quad P(A\cap B)= \frac{2}{6}, \quad P(B) = \frac{3}{6}, \quad P(A|B) = \frac{2/6}{3/6}=\frac{2}{3}.

📘Regla del producto

\textbf{📌 Regla del producto (para dos eventos)}
\text{Sean} \ A,B \in \mathcal{A}, \ \text{con} \ P(A), P(B)>0. \ \text{Entonces:}
P(A \cap B) = P(A|B)P(B) = P(B|A)P(A).
\textbf{🔧 Extensión a tres eventos}
\text{Sean} \ A,B,C \in \mathcal{A}, \ \text{con probabilidades bien definidas. Entonces:}
\textbf{📊 Ejemplo:}
\text{Se extraen dos cartas sin reemplazo de un mazo estándar y definimos}
\text{🔹 }A=\{\text{``la primera carta es un as"}\}
P(A) = \frac{4}{52}, \quad P(B|A) = \frac{3}{51}
P(A \cap B \cap C) = P(A)P(B|A)P(C|A\cap B).
\text{🔹 }B=\{\text{``la segunda carta es un as"}\}
P(A\cap B) = \frac{4}{52}\cdot \frac{3}{51} = \frac{1}{121}.

📘Independencia de eventos

\textbf{📌 Definición (Independencia)}
\text{Sean} \ A,B \in \mathcal{A}. \ \text{Se dice que} \ A \ \text{y} \ B \ \text{son independientes si}
P(A \cap B) = P(A)P(B)
\textbf{🔧 Independencia mutua (más de dos eventos)}
\text{Eventos} \ A_1,A_2,\ldots,A_n \ \text{son mutuamente independientes si para toda subcolección}
P(A_{i_1} \cap \cdots \cap A_{i_k}) = P(A_{i_1}) \cdots P(A_{i_k})
\{A_{i_1}, \dots, A_{i_k}\} \ \text{se cumple:}
\text{❗ Independencia por pares no implica independencia mutua.}
\textbf{🟨 Interpretación}
\text{Conocer que ocurrió} \ B \ \text{no cambia la probabilidad de} \ A, \ \text{y viceversa:}
P(A \mid B) = P(A) \quad \text{y} \quad P(B \mid A) = P(B)

📘Independencia de eventos

\textbf{📌 Definición (Independencia)}
\text{Sean} \ A,B \in \mathcal{A}. \ \text{Se dice que} \ A \ \text{y} \ B \ \text{son independientes si}
P(A \cap B) = P(A)P(B)
\textbf{🔧 Independencia mutua (más de dos eventos)}
\text{Eventos} \ A_1,A_2,\ldots,A_n \ \text{son mutuamente independientes si para toda subcolección}
P(A_{i_1} \cap \cdots \cap A_{i_k}) = P(A_{i_1}) \cdots P(A_{i_k})
\{A_{i_1}, \dots, A_{i_k}\} \ \text{se cumple:}
\text{❗ Independencia por pares no implica independencia mutua.}
\textbf{🟨 Interpretación}
\text{Conocer que ocurrió} \ B \ \text{no cambia la probabilidad de} \ A, \ \text{y viceversa:}
P(A \mid B) = P(A) \quad \text{y} \quad P(B \mid A) = P(B)

📘Eventos mutuamente excluyentes

\textbf{📌 Definición}
\text{Se dice que dos eventos} \ A,B \in \mathcal{A} \ \text{son mutuamente excluyentes si:}
A \cap B = \emptyset \Rightarrow P(A \cap B) = 0
\textbf{🧪 Ejemplo}
\text{Lanzar un dado}
\text{🔹 }A=\{\text{Sale }2 \text{ o }4\}
\text{🔹 }B=\{\text{Sale }5 \text{ o }6\}
\Rightarrow
A \cap B=\emptyset, \ \text{y por lo tanto} \ P(A \cup B) = P(A)+P(B)
\textbf{🔧 Mutuamente excluyentes por pares}
\text{Una colección de eventos} \ A_1,A_2,\ldots,A_n \ \text{es mutuamente excluyente por pares si:}
A_i \cap A_j = \emptyset \quad \text{para todo } i \ne j
\text{🧩 Si además } \bigcup_{i=1}^{n}A_i = \Omega, \ \text{se dice que forman una partición de }\Omega.

📘Eventos mutuamente excluyentes

\textbf{❗ Recordatorio importante}
\text{No confundir independencia con exclusión mutua}
Concepto Condición ¿Pueden ocurrir simultáneamente?
Mutuamente excluyentes ❌ Nunca ocurren juntas
Independientes ✅ Puede ocurrir ambas
A \cap B = \emptyset
P(A \cap B) = P(A)P(B)
\textbf{🧪 Ejemplo}
\text{Si } A \ \text{y} \ B \ \text{son mutuamente excluyentes y ambos tienen probabilidad positiva,}
\text{entonces:}
P(A \cap B) = 0 \ne P(A)P(B)
\text{es decir, no pueden ser independientes.}

📘Partición del espacio muestral

\textbf{📌 Definición (Partición del espacio muestral)}
\text{Sea} \ \Omega \ \text{un espacio muestral, y sea }\{A_1,A_2,\ldots,A_n\}\subseteq \mathcal{A}. \ \text{Decimos que esta colección}
\text{de eventos forma una partición de }\Omega \ \text{si se cumplen:}
\text{1.} \ A_i\cap A_j = \emptyset, \quad \forall i\ne j
\text{2.} \ \bigcup_{i=1}^{n}A_i=\Omega
\textbf{📊 Ejemplo}
\text{Sea } \ A \in \mathcal{A}, \ \text{entonces } \{A, A^c\} \ \text{es una partición de }\Omega.
\text{Suponga que se lanza un dado. Sea }A=\{\text{``número par"}\}. \ \text{Entonces: }
A=\{2,4,6\}; \quad A^c=\{1,3,5\} \Rightarrow \{A,A^c\} \ \text{es una partición de }\Omega=\{1,2,3,4,5,6\}
\textbf{🧩 Notas}
\text{🔹 Las particiones permiten descomponer probabilidades en partes más simples.}

📘Teoremas principales

\textbf{📌 Teorema de la Probabilidad Total}
\text{Sea }\{B_1,B_2,\ldots,B_n\}\subseteq\mathcal{A}, \ \text{una partición de }\Omega, \ \text{tal que }P(B_i)>0 \ \forall_i, \ \text{entonces,}
\text{para cualquier evento } A \in \mathcal{A}, \ \text{se cumple:}
P(A)=\sum_{i=1}^{n}P(A\mid B_i)\cdot P(B_i).
\textbf{📌 Teorema de Bayes}
\text{Sea }\{B_1,B_2,\ldots,B_n\}\subseteq\mathcal{A}, \ \text{una partición de }\Omega, \ \text{tal que }P(B_i)>0, \ \text{y sea, }A \in \mathcal{A} \ \text{con}
P(A)>0. \ \text{Entonces, para cada }i=1,2,\ldots,n:
P(B_i\mid A) = \frac{P(A\mid B_i)P(B_i)}{\sum_{j=1}^{n}P(A|B_j)P(B_j)}.
