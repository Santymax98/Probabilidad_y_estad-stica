📘Variable aleatoria

\textbf{📌 Definición (Variable aleatoria)}
\text{Sea} \ (\Omega, \mathcal{A}, P), \ \text{un espacio de probabilidad. Una función} \ X: \Omega \to \mathbb{R} \ \text{se denomina variable}
\forall B \in \mathcal{B}(\mathbb{R}), \quad X^{-1}(B) \in \mathcal{A}
\text{aleatoria si es medible con respecto a} \ \mathcal{A} \ \text{y la} \ \sigma-\text{álgebra de Borel} \ \mathcal{B}(\mathbb{R}), \ \text{es decir:}
\text{Equivalente a:}
\forall x \in \mathbb{R}, \quad \{\omega \in \Omega: X(\omega) \leq x \} \in \mathcal{A}
\text{🧠 Interpretación:}
\text{🔹 }X \ \text{asigna un número real a cada resultado aleatorio} \ \omega \in \Omega.
\text{🔹 } \ \text{La condición de medibilidad garantiza que podemos medir eventos del tipo}
X\leq x \ \text{usando la probabilidad} \ P.

📘Variable aleatoria

\textbf{📌 Variable aleatoria discreta}
\text{Una variable aleatoria} \ X \ \text{es discreta si su imagen es un conjunto finito o numerable:}
\text{Im} (X) = \{x_1, x_2, \ldots,\} \subseteq \mathbb{R}
\text{💡 En este caso, existe una función de masa de probabilidad (f.m.p.):}
p_X(x) = P(X = x), \quad \text{tal que } \sum\limits_{x \in \operatorname{Im}(X)} p_X(x) = 1
\textbf{🧩 Ejemplos}
\text{🔹 }\text{Número de hijos en una familia → valores:} \ \{0,1,2,\ldots\}
\text{🔹 }\text{Lanzar una moneda equilibrada → valores:} \ \{0,1\}

📘Variable aleatoria

\textbf{📌 Variable aleatoria Continua}
\text{Una variable aleatoria} \ X \ \text{es continua si existe una función de densidad} \ f_X: \mathbb{R} \to [0, \infty)
\text{tal que:}
P(a \leq X \leq b) = \int\limits_{a}^{b} f_X(x)\, dx
\text{y se cumple}
\textbf{🧩 Ejemplos}
\text{🔹 }\text{altura de una persona seleccionada al azar → valores: intervalo en} \ \mathbb{R}
\text{🔹 }\text{Lanzar un dardo dentro del cuadrado unitario → valores:} \ [-1,1]\times[-1,1]
\int\limits_{-\infty}^{\infty} f_X(x)\, dx = 1

📘Variable aleatoria

\textbf{📌 Función de distribución acumulada}
\text{Sea} \ X \ \text{una variable aleatoria. La función de distribución acumulada asociada a} \ X \ \text{es la}
\text{función} \ F_X: \mathbb{R} \to [0,1] \ \text{definida por:}
F_X(x) = P(X \leq x)
\textbf{🧠 Propiedades generales de} \ F_X
\text{1. Monótona no decreciente:} \quad x_1 \leq x_2 \Rightarrow F_X(x_1) \leq F_X(x_2)
\text{2. Límites en los extremos:} \quad \lim_{x \to -\infty} F_X(x) = 0, \quad \lim_{x \to +\infty} F_X(x) = 1
\text{3. Continua por la derecha:} \quad \forall x \in \mathbb{R}, \quad \lim_{h \downarrow 0} F_X(x + h) = F_X(x)
\text{4. Saltos:} \quad P(X = x) = F_X(x) - \lim_{h \uparrow 0} F_X(x - h) \quad \text{(discretas)}

📘Valor Esperado

\textbf{📌 Definición (Valor esperado o esperanza matemática)}
\text{Sea} \ X \ \text{una variable aleatoria}
\text{🔹 Si }X \ \text{es discreta con función de probabilidad }p(x):
\mathbb{E}[X] = \sum\limits_{\chi}x\cdot p(x)
\text{🔹 Si }X \ \text{es continua con función de densidad }f(x):
\mathbb{E}[X] = \int\limits_{\chi}x\cdot f(x)dx
\text{🟨 Interpretación}
\text{🔹 Representa el promedio teórico de los valores de }X
\text{🔹 Si repites el experimento muchas veces, el promedio de los resultados se}
\text{aproxima a }\mathbb{E}[X]

📘Valor Esperado

\textbf{📌 Proposición}
\text{Sea} \ X \ \text{una variable aleatoria y }g:\mathbb{R}\to \mathbb{R} \ \text{una función tal que }\mathbb{E}[|g(x)|] < \infty. \ \text{Entonces}
\text{🔹 Si }X \ \text{es discreta con función de probabilidad }p(x):
\mathbb{E}[g(X)] = \sum\limits_{\chi}g(x)\cdot p(x)
\text{🔹 Si }X \ \text{es continua con función de densidad }f(x):
\mathbb{E}[g(X)] = \int\limits_{\chi}g(x)\cdot f(x)dx
\text{🟨 Interpretación}
\text{🔹 Esta propiedad generaliza el valor esperado, permitiendo calcular esperanzas}
\text{🔹 Es fundamental para definir momentos: } \mathbb{E}[X^2], \mathbb{E}[X^3]...
\text{de transformaciones de }X, \ \text{como} \ X^2, \ln(X), e^X, \ \text{etc.}

📘Valor Esperado

\textbf{📌 Propiedades}
\text{Sean} \ X \ \text{y} \ Y \ \text{dos variables aleatorias con }\mathbb{E}[X], \mathbb{E}[Y] < \infty, \ \text{y sea} \ c \in \mathbb{R}. \ \text{Entonces se}
\text{1. } \mathbb{E}[c]=c
\text{Estas propiedades hacen del valor esperado una herramienta algebraicamente}
\text{robusta. Son fundamentales para el desarrollo del cálculo estocástico, inferencia}
\text{y teoría de estimadores.}
\text{cumplen las siguientes propiedades:}
\text{2. } \mathbb{E}[cX]=c\cdot \mathbb{E}[X]
\text{3. Si }X \geq 0 \ \text{entonces, } \mathbb{E}[X] \geq 0
\text{4. }\mathbb{E}[X + Y] = \mathbb{E}[X] + \mathbb{E}[Y]
\text{5. Si }X \ \text{y} \ Y \ \text{son independientes, entonces } \mathbb{E}[XY]=\mathbb{E}[X]\mathbb{E}[Y].
\text{🧠 Nota.}

📘Varianza

\textbf{📌 Definición: Varianza}
\text{Sean} \ X \ \text{una variable aleatoria con} \ \mathbb{E}[X]=\mu. \ \text{La varianza de} \ X \ \text{cumplen las siguientes}
\text{1. } \text{Var}(c)>0
\text{Estas propiedades hacen de la varianza una medida robusta y operacionalmente}
\text{útil especialmente en combinaciones lineales, análisis de errores y propagación}
\text{de incertidumbre.}
\text{propiedades:}
\text{2. } \text{Var}(c)=0
\text{3. Var}(cX) = c \text{Var}(X)
\text{4. Var}(X + c) = \text{Var}(X)
\text{5. Var}(X)=\mathbb{E}[X^2]-\mathbb{E}[X]^2
\text{🧠 Nota.}
